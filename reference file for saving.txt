function normalizeSkipConfigSpec(array $entry) {
    $lc         = (isset($entry['lotteryConfig']) && is_array($entry['lotteryConfig']))
                    ? $entry['lotteryConfig']
                    : [];
    $hasExtra   = !empty($lc['has_extra_ball']) && isset($lc['num_extra_balls_drawn'])
                    && (int) $lc['num_extra_balls_drawn'] > 0;
    $extraCount = $hasExtra ? (int) $lc['num_extra_balls_drawn'] : 0;
    return [
        'gameId'         => isset($entry['gameId'])                    ? (string) $entry['gameId']                    : '',
        'gameName'       => isset($entry['lotteryName'])               ? (string) $entry['lotteryName']               : '',
        'tableName'      => isset($entry['dbCol'])                     ? (string) $entry['dbCol']                     : '',

/**
 * Retrieves the next non-blank next_draw_date, expanding #__ prefix.
 * Returns DATETIME (adds ' 00:00:00' if the source is DATE-only).
 */
function getNextDrawDate($gameId, $db, $tableName) {
    $tbl = $db->quoteName($db->replacePrefix($tableName));
    $q   = $db->getQuery(true)
        ->select($db->quoteName('next_draw_date'))
        ->from($tbl)
        ->where($db->quoteName('game_id') . ' = ' . $db->quote($gameId))
        ->where($db->quoteName('next_draw_date') . ' IS NOT NULL')
        ->where($db->quoteName('next_draw_date') . " <> " . $db->quote('0000-00-00'))
        ->where($db->quoteName('next_draw_date') . " <> " . $db->quote('0000-00-00 00:00:00'))
        ->order($db->quoteName('draw_date') . ' DESC')
        ->setLimit(1);
    $db->setQuery($q);
    $val = $db->loadResult();

    // Normalize DATE to DATETIME for inserts downstream
    if (!empty($val) && preg_match('/^\d{4}-\d{2}-\d{2}$/', $val)) {
        $val .= ' 00:00:00';
    }
    return $val ?: null;
}

function leJsonResponse(bool $success, string $message, array $extra = []): void
{
    if (!headers_sent()) {

    }

    echo json_encode(
        array_merge(

                'success' => $success,
                'message' => $message,
            ],
            $extra
        ),
        JSON_UNESCAPED_SLASHES | JSON_UNESCAPED_UNICODE
    );

    if (class_exists('Joomla\\CMS\\Factory')) {
        try {
            \Joomla\CMS\Factory::getApplication()->close();
        } catch (Throwable $e) {
            exit;
        }
    }

    exit;
}

function leCleanText(string $value, int $maxLength = 120): string
{
    $value = trim(strip_tags($value));
    $value = preg_replace('/\s+/', ' ', $value) ?: '';

    if (function_exists('mb_substr')) {
        return mb_substr($value, 0, $maxLength, 'UTF-8');
    }

    return substr($value, 0, $maxLength);
}

function leCleanNumberList(string $value, int $maxItems, int $maxNumber, bool $allowEmpty = false): string
{
    $value = trim($value);

    if ($value === '') {
        return $allowEmpty ? '' : '';
    }

    if (!preg_match('/^\d+(,\d+)*$/', $value)) {
        return '';
    }

    $parts = explode(',', $value);
    $clean = [];
    $seen  = [];

    foreach ($parts as $part) {
        $num = (int) $part;

        if ($num < 1 || $num > $maxNumber) {
            return '';
        }

        if (isset($seen[$num])) {
            continue;
        }

        $seen[$num] = true;
        $clean[]    = (string) $num;

        if (count($clean) > $maxItems) {
            return '';
        }
    }

    if (!$allowEmpty && count($clean) === 0) {
        return '';
    }

    return implode(',', $clean);
}

function leClampInt($value, int $min, int $max, int $fallback): int
{
    $value = filter_var($value, FILTER_VALIDATE_INT);

    if ($value === false || $value < $min || $value > $max) {
        return $fallback;
    }

    return (int) $value;
}

function leClampFloat($value, float $min, float $max, float $fallback): float
{
    $value = filter_var($value, FILTER_VALIDATE_FLOAT);

    if ($value === false || $value < $min || $value > $max) {
        return $fallback;
    }

    return (float) $value;
}

function leCleanDateTimeForDb(string $value): string
{
    $value = trim($value);

    if (preg_match('/^\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}$/', $value)) {
        return $value;
    }

    return date('Y-m-d H:i:s');
}


function leCleanDateForDbNullable(string $value): ?string
{
    $value = trim($value);

    if ($value === '') {
        return null;
    }

    if (preg_match('/^\d{4}-\d{2}-\d{2}$/', $value)) {
        return $value . ' 00:00:00';
    }

    if (preg_match('/^\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}$/', $value)) {
        return $value;
    }

    return null;
}

function leCleanHash(string $value): string

function leCleanRunUuid(string $value): string

}

function leNormalizeDecimalForHash(float $value, int $precision): string
{
    $out = number_format($value, $precision, '.', '');
    $out = rtrim(rtrim($out, '0'), '.');
    return $out === '' ? '0' : $out;
}

function leGetTableColumns($db, string $tableName): array
{
    static $cache = [];
    $resolved = $db->replacePrefix($tableName);

    if (isset($cache[$resolved])) {
        return $cache[$resolved];
    }

    try {
        $db->setQuery('SHOW COLUMNS FROM ' . $db->quoteName($resolved));
        $rows = $db->loadObjectList();
    } catch (Throwable $e) {
        error_log('[AI-Lotteries] Could not inspect table columns for ' . $resolved . ': ' . $e->getMessage());
        $cache[$resolved] = [];
        return [];
    }

    $columns = [];
    foreach ($rows as $row) {
        if (isset($row->Field)) {
            $columns[(string) $row->Field] = true;
        }
    }

    $cache[$resolved] = $columns;
    return $columns;
}

// (A) Helper to compute “n choose k”
function combination($n, $k) {
    if ($k > $n) return 0;
    $res = 1;
    for ($i = 1; $i <= $k; $i++) {
        $res = $res * ($n - $i + 1) / $i;
    }
    return round($res);
}

// (B) Total ways to pick $pickSize from $mainNumbersMax

// Utility: build a tensor of recency-decay weights
function buildSampleWeightTensor(numSamples, decay) {
  var weights = [];
  for (var i = 0; i < numSamples; i++) {
    weights.push(Math.pow(decay, (numSamples - 1) - i));
  }
  return tf.tensor1d(weights);
}

// NOTE: SKIP_ALPHA remains for backwards-compatibility, but SKAI now uses a coherent expert blend.
var SKAI_EPS               = 1e-9;
var SKAI_SKIP_BETA_DEFAULT = 2.0;  // controls how strongly skipScores separate numbers (hoist-safe)
var SKAI_BLEND_DEFAULT     = 0.65; // Balanced-mode default AI weight λ (hoist-safe)

// Mode → λ (AI weight). Higher λ = more AI-forward.
function getSkaiBlendLambda(modeKey) {
  if (modeKey === 'ai-forward')     { return 0.90; }
  if (modeKey === 'conservative')   { return 0.80; }
  if (modeKey === 'explorative')    { return 0.45; }
  if (modeKey === 'skip-dominant')  { return 0.35; }
  // mixed is adaptive; computed from ai/skip distributions (see getSkaiLambda below)
  if (modeKey === 'mixed')          { return NaN; }
  // balanced / unknown
  return SKAI_BLEND_DEFAULT;
}

// Jensen-Shannon divergence for two discrete distributions (bounded, symmetric, stable).
function skaiJSDivergence(p, q) {
  var n = Math.min(p.length, q.length);
  var js = 0;
  for (var i = 0; i < n; i++) {
    var pi = Math.max(SKAI_EPS, p[i]);
    var qi = Math.max(SKAI_EPS, q[i]);
    var mi = 0.5 * (pi + qi);
    js += 0.5 * (pi * Math.log(pi / mi) + qi * Math.log(qi / mi));
  }
  // Natural log units; bounded in [0, ln(2)]
  return js;
}

// Mixed mode: λ adapts to how much AI and Skip agree.
// - If they agree (low JSD), lean AI-forward more.
// - If they disagree (high JSD), pull λ toward 0.5 to avoid over-committing.
function skaiAdaptiveLambdaMixed(aiNorm, skipProb) {
  var js = skaiJSDivergence(aiNorm, skipProb);
  var jsMax = Math.log(2); // theoretical maximum
  var agree = 1 - Math.min(1, js / jsMax); // 1=agree, 0=disagree

  // Map agreement into a safe λ band for Mixed:
  // - High agreement → λ up to ~0.80
  // - Low agreement  → λ down to ~0.50
  var lambda = 0.50 + (0.30 * agree);

  // Clamp defensively
  return Math.max(0.35, Math.min(0.90, lambda));
}

// Unified λ getter: fixed for other modes; adaptive for Mixed.
function getSkaiLambda(modeKey, aiNorm, skipProb) {
  if (modeKey === 'mixed') {
    return skaiAdaptiveLambdaMixed(aiNorm, skipProb);
  }
  var lam = getSkaiBlendLambda(modeKey);
  return (typeof lam === 'number' && isFinite(lam)) ? lam : SKAI_BLEND_DEFAULT;
}

// Mode-aware skip separation (beta). Higher beta makes Skip expert "sharper".
function getSkaiSkipBeta(modeKey) {
  if (modeKey === 'skip-dominant') { return 3.0; }
  if (modeKey === 'explorative')   { return 2.5; }
  if (modeKey === 'conservative')  { return 1.6; }
  if (modeKey === 'ai-forward')    { return 1.6; }
  if (modeKey === 'mixed')         { return 2.2; }
  return SKAI_SKIP_BETA_DEFAULT; // balanced
}

// Temperature for exploration sampling (τ >= 1). τ=1 means no flattening.
function getSkaiExploreTau(modeKey) {
  if (modeKey === 'explorative') { return 1.8; }
  if (modeKey === 'mixed')       { return 1.4; }
  return 1.0;
}

// How many picks should be "explore sampled" instead of strict top-k.
// All modes get at least 1 explore pick to prevent identical outputs across batch runs.
function getSkaiExploreCount(modeKey, pickSize) {
  if (modeKey === 'explorative')   { return Math.max(1, Math.floor(pickSize * 0.40)); }
  if (modeKey === 'mixed')         { return Math.max(1, Math.floor(pickSize * 0.30)); }
  if (modeKey === 'skip-dominant') { return Math.max(1, Math.floor(pickSize * 0.20)); }
  // All other modes (ai-forward, balanced, conservative): 1 exploration pick minimum
  return 1;
}

// Build a tempered mass (flattened) for sampling: p^(1/τ), normalized.
function skaiTemperedMass(probMass, tau) {
  if (!tau || tau <= 1) {
    return normalizePositive(probMass);
  }
  var raw = new Array(probMass.length);
  var inv = 1 / tau;
  for (var i = 0; i < probMass.length; i++) {
    raw[i] = Math.pow(Math.max(SKAI_EPS, probMass[i]), inv);
  }
  return normalizePositive(raw);
}

// Weighted sampling without replacement from a probability mass.
function skaiSampleWithoutReplacement(probMass, count) {
  var picks = [];
  var available = probMass.slice(0);

  for (var k = 0; k < count; k++) {
    // Build CDF
    var sum = 0;
    for (var i = 0; i < available.length; i++) {
      sum += Math.max(0, available[i]);
    }
    if (sum <= 0) {
      break;
    }

    var r = Math.random() * sum;
    var chosen = -1;
    for (var i = 0; i < available.length; i++) {
      r -= Math.max(0, available[i]);
      if (r <= 0) { chosen = i; break; }
    }
    if (chosen < 0) { break; }

    picks.push(chosen);
    available[chosen] = 0; // remove
  }

  return picks;
}

// Select final picks: exploit top-(pickSize-exploreCount), explore sample the rest.
function skaiSelectPicksFromMass(probMass, pickSize, modeKey) {
  var exploreCount = getSkaiExploreCount(modeKey, pickSize);
  var tau = getSkaiExploreTau(modeKey);

  var indices = [];
  // Exploit indices by sorting
  var ranked = probMass.map(function(p, i) { return { i: i, p: p }; }).sort(function(a, b) { return b.p - a.p; });

  var exploitCount = Math.max(0, pickSize - exploreCount);
  for (var k = 0; k < exploitCount && k < ranked.length; k++) {
    indices.push(ranked[k].i);
  }

  if (exploreCount > 0) {
    // Build remaining mass excluding already-picked exploit indices
    var remaining = probMass.slice(0);
    for (var k = 0; k < indices.length; k++) {
      remaining[indices[k]] = 0;
    }

    var tempered = skaiTemperedMass(remaining, tau);
    var sampled = skaiSampleWithoutReplacement(tempered, exploreCount);
    for (var k = 0; k < sampled.length; k++) {
      indices.push(sampled[k]);
    }
  }

  // Return sorted numeric values (1-based numbers) for stable display (dedup defensive)
  var seen = {};
  var uniq = [];
  for (var m = 0; m < indices.length; m++) {
    var idx = indices[m] + 1;
    if (!seen[idx]) { seen[idx] = true; uniq.push(idx); }
  }
  return uniq.sort(function(a, b) { return a - b; });
}

// Normalize a positive vector into sum-to-1 mass (with epsilon floor).
function normalizePositive(arr) {
  var sum = 0;
  for (var i = 0; i < arr.length; i++) {
    var v = (typeof arr[i] === 'number' && isFinite(arr[i])) ? Math.max(SKAI_EPS, arr[i]) : SKAI_EPS;
    sum += v;
  }
  var out = new Array(arr.length);
  var denom = (sum > 0) ? sum : (arr.length * SKAI_EPS);
  for (var i = 0; i < arr.length; i++) {
    var v = (typeof arr[i] === 'number' && isFinite(arr[i])) ? Math.max(SKAI_EPS, arr[i]) : SKAI_EPS;
    out[i] = v / denom;
  }
  return out;
}

// Turn skipScores (0..1) into a proper expert distribution using a softmax-like transform.
function buildSkipExpertProbs(skipScores, beta) {
  var raw = new Array(skipScores.length);
  for (var i = 0; i < skipScores.length; i++) {
    var u = (typeof skipScores[i] === 'number' && isFinite(skipScores[i])) ? skipScores[i] : 0;
    // center at 0.5 to reduce runaway bias; beta controls separation
    raw[i] = Math.exp(beta * (u - 0.5));
  }
  return normalizePositive(raw);
}

// Log-opinion pool (geometric mixture): p ∝ ai^λ * skip^(1-λ)
function logOpinionPool(aiProbs, skipProbs, lambda) {
  var out = new Array(aiProbs.length);
  for (var i = 0; i < aiProbs.length; i++) {
    var a = Math.max(SKAI_EPS, aiProbs[i]);
    var s = Math.max(SKAI_EPS, skipProbs[i]);
    out[i] = Math.exp(lambda * Math.log(a) + (1 - lambda) * Math.log(s));
  }
  return normalizePositive(out);
}

// --- SKAI backtest proper scoring rules (math-correct evaluation) ---
function skaiClipProb(p) {
  if (typeof p !== 'number' || !isFinite(p)) { return SKAI_EPS; }
  return Math.min(1 - SKAI_EPS, Math.max(SKAI_EPS, p));
}

function skaiBrierScore(probMass, actualSet, maxVal) {
  // probMass: array length maxVal, sum-to-1
  // actualSet: Set of winning numbers in 1..maxVal
  var sum = 0;
  for (var i = 0; i < maxVal; i++) {
    var p = skaiClipProb(probMass[i]);
    var y = skaiSetHas(actualSet, i + 1) ? 1 : 0;
    var d = p - y;
    sum += d * d;
  }
  return sum / maxVal;
}

function skaiLogLoss(probMass, actualSet, maxVal) {
  // Strictly proper for Bernoulli marginals; clipped to avoid -Inf
  var sum = 0;
  for (var i = 0; i < maxVal; i++) {
    var p = skaiClipProb(probMass[i]);
    var y = skaiSetHas(actualSet, i + 1) ? 1 : 0;
    sum += y ? -Math.log(p) : -Math.log(1 - p);
  }
  return sum / maxVal;
}

function skaiExpectedMassInPicks(probMass, picks) {
  // picks: array of numbers (1..max)
  var s = 0;
  for (var i = 0; i < picks.length; i++) {
    var idx = picks[i] - 1;
    if (idx >= 0 && idx < probMass.length) {
      s += skaiClipProb(probMass[idx]);
    }
  }
  return s;
}

// Compute per-draw Skip feature vectors for a chronological series of draws.
// drawsNumbers: [ [n1, n2, ...], ... ] in OLDEST → NEWEST order.
// For each draw t, we compute skipVec[t][k] = draws since number (k+1) last appeared BEFORE draw t.
function computeSkipFeatureSeries(drawsNumbers, pickSize, mainNumbersMax) {
  var lastHitIndex   = skaiArrayFill(mainNumbersMax, null);
  var skipsPerStep   = [];

  for (var t = 0; t < drawsNumbers.length; t++) {
    var skipVec = skaiArrayFill(mainNumbersMax, 0);

    // For each number, how long since last seen?
    for (var k = 0; k < mainNumbersMax; k++) {
      var last = lastHitIndex[k];
      // If we've never seen it, treat skip as t (all previous draws missed it).
      var skip = (last === null) ? t : (t - last - 1);
      skipVec[k] = skip;
    }

    // Now update lastHitIndex using the CURRENT draw's main numbers.
    var draw = drawsNumbers[t];
    for (var j = 0; j < pickSize; j++) {
      var n = draw[j];
      if (n >= 1 && n <= mainNumbersMax) {
        lastHitIndex[n - 1] = t;
      }
    }

    skipsPerStep.push(skipVec);
  }

  // Normalize skips to 0..1 so they play nicely with normalized inputs.
  var maxSkip = 0;
  for (var t = 0; t < skipsPerStep.length; t++) {
    var vec = skipsPerStep[t];
    for (var i = 0; i < vec.length; i++) {
      if (vec[i] > maxSkip) {
        maxSkip = vec[i];
      }
    }
  }

  if (maxSkip > 0) {
    for (var t = 0; t < skipsPerStep.length; t++) {
      var vec = skipsPerStep[t];
      for (var i = 0; i < vec.length; i++) {
        vec[i] = vec[i] / maxSkip;
      }
    }
  }

  return skipsPerStep;
}

// Compute per-number Skip scores (0..1) from a DESC list of draws (most recent first).
// This is a thin wrapper around computeSkipFeatureSeries so older code that calls
// computeSkipScores(...) continues to work.
function computeSkipScores(drawsDesc, pickSize, mainNumbersMax) {
  // Convert DESC draws into ASC array of numbers for computeSkipFeatureSeries.
  var drawsAscNumbers = drawsDesc.slice().reverse().map(function(d) { return d.numbers; });
  if (!drawsAscNumbers.length) {
    return skaiArrayFill(mainNumbersMax, 0);
  }

  var skipSeries = computeSkipFeatureSeries(drawsAscNumbers, pickSize, mainNumbersMax);
  var latestVec  = skipSeries[skipSeries.length - 1] || [];
  var result     = skaiArrayFill(mainNumbersMax, 0);

  for (var i = 0; i < mainNumbersMax; i++) {
    result[i] = (typeof latestVec[i] === 'number') ? latestVec[i] : 0;
  }

  return result;
}

function trainModelForBacktest(trainingDraws, settings, onEpochEndCallback) {
  var pickSize       = settings.PICK_SIZE;
  var mainNumbersMax = settings.MAIN_MAX;
  var extraBallMax   = settings.EXTRA_MAX;
  var hasExtraBall   = settings.HAS_EXTRA;

  // 1) normalize inputs
  var drawsNumbers = trainingDraws.map(function(d) { return d.numbers; });
  var normalized   = drawsNumbers.map(function(arr) {
    return arr.map(function(n, i) {
      return i < pickSize
        ? n / mainNumbersMax
        : (hasExtraBall ? n / extraBallMax : n);
    });
  });

  // 1b) compute Skip feature series (per draw, per main number)
  var skipFeatureSeries = computeSkipFeatureSeries(drawsNumbers, pickSize, mainNumbersMax);

  // 2) build X and Y arrays
  var xs = [], ys_main = [], ys_extra = [];
  for (var i = 0; i < normalized.length - 1; i++) {
    var baseInput    = normalized[i];
    var skipFeatures = skipFeatureSeries[i];
    var inputVec     = baseInput.concat(skipFeatures);
    xs.push(inputVec);

    var nextNums = drawsNumbers[i + 1];

    // Y for main
    var mainVec = skaiArrayFill(mainNumbersMax, 0);
    nextNums.slice(0, pickSize).forEach(function(n) { mainVec[n - 1] = 1; });
    ys_main.push(mainVec);

    // Y for extra (supports 1 or more extra balls)
    if (hasExtraBall) {
      var extraVec = skaiArrayFill(extraBallMax, 0);
      var _numExtra = settings.EXTRA_COUNT || 1;
      for (var _eb2 = 0; _eb2 < _numExtra; _eb2++) {
        var _en = nextNums[pickSize + _eb2];
        if (_en >= 1 && _en <= extraBallMax) {
          extraVec[_en - 1] = 1;
        }
      }
      ys_extra.push(extraVec);
    }
  }

  // 3) create tensors
  var xsTensor     = tf.tensor2d(xs);
  var ysMainTensor = tf.tensor2d(ys_main);
  var ysExtraTensor;
  if (hasExtraBall) {
    ysExtraTensor = tf.tensor2d(ys_extra);
  }

  // 4) recency-decay weights
  var sampleWeightTensor = buildSampleWeightTensor(xs.length, settings.RECENCY_DECAY);

  // 5) build the model
  var input = tf.input({ shape: [xs[0].length] });
  var x = tf.layers.dense({ units: 128, activation: settings.ACTIVATION_FUNCTION }).apply(input);
  x = tf.layers.dropout({ rate: settings.DROPOUT_RATE }).apply(x);
  for (var i = 0; i < settings.HIDDEN_LAYERS; i++) {
    x = tf.layers.dense({ units: 64, activation: settings.ACTIVATION_FUNCTION }).apply(x);
    x = tf.layers.dropout({ rate: settings.DROPOUT_RATE }).apply(x);
  }
  var mainOut = tf.layers.dense({
    units: settings.MAIN_MAX,
    activation: 'sigmoid',
    name: 'main_numbers_output'
  }).apply(x);

  var model, outputs;
  if (hasExtraBall) {
    var extraOut = tf.layers.dense({
      units: settings.EXTRA_MAX,
      activation: 'sigmoid',
      name: 'extra_output'
    }).apply(x);
    outputs = [mainOut, extraOut];
    model = tf.model({ inputs: input, outputs: outputs });
  } else {
    outputs = mainOut;
    model = tf.model({ inputs: input, outputs: outputs });
  }

  // 6) compile
  var optimizer = tf.train.adam(settings.LEARNING_RATE);
  var loss      = hasExtraBall
    ? ['binaryCrossentropy','binaryCrossentropy']
    : 'binaryCrossentropy';
  model.compile({ optimizer: optimizer, loss: loss, metrics: ['accuracy'] });

  // 7) prepare yTrain (no sampleWeight support in this tfjs build)
  var yTrain;
  if (hasExtraBall) {
    yTrain = {
      main_numbers_output: ysMainTensor,
      extra_output: ysExtraTensor
    };
  } else {
    yTrain = ysMainTensor;
  }

  // 8) fit model (recency-weighted)
  // sampleWeightTensor is a per-sample exponential decay weight (newest samples weigh more).
  var fitArgs = {
    epochs:          settings.EPOCHS,
    batchSize:       settings.BATCH_SIZE,
    shuffle:         false,
    validationSplit: 0.2,
    callbacks:       { onEpochEnd: onEpochEndCallback }
  };

          // PATCH: sampleWeight not supported in this TFJS build; disable to avoid runtime error.
          // Recency is still used in backtesting via SKAI blend + skip features; full per-sample weighting requires a TFJS build that supports sampleWeight.
          // fitArgs.sampleWeight = hasExtraBall
          //   ? [sampleWeightTensor, sampleWeightTensor]
          //   : sampleWeightTensor;

  return model.fit(xsTensor, yTrain, fitArgs).then(function() {
    // Dispose training tensors after fit; the trained model is returned for prediction.
    try {
      if (xsTensor && typeof xsTensor.dispose === "function") { xsTensor.dispose(); }
      if (ysMainTensor && typeof ysMainTensor.dispose === "function") { ysMainTensor.dispose(); }
      if (ysExtraTensor && typeof ysExtraTensor.dispose === "function") { ysExtraTensor.dispose(); }
      if (sampleWeightTensor && typeof sampleWeightTensor.dispose === "function") { sampleWeightTensor.dispose(); }
    } catch (btTrainCleanupError) {
      console.warn("Backtest training tensor cleanup warning:", btTrainCleanupError);
    }
    return model;
  });
}

  function formToEncodedBody(form) {
    var data = [];
    var inputs = form.elements;

    for (var i = 0; i < inputs.length; i++) {
      var el = inputs[i];

      if (!el.name) {
        continue;
      }

      var tag = el.tagName.toUpperCase();

      if (tag === "INPUT") {
        if ((el.type === "checkbox" || el.type === "radio") && !el.checked) {
          continue;
        }

        data.push(encodeURIComponent(el.name) + "=" + encodeURIComponent(el.value || ""));
      } else if (tag === "TEXTAREA" || tag === "SELECT") {
        data.push(encodeURIComponent(el.name) + "=" + encodeURIComponent(el.value || ""));
      } else if (tag === "BUTTON" && el.type === "submit" && el.name) {
        data.push(encodeURIComponent(el.name) + "=" + encodeURIComponent(el.value || "1"));
      }
    }

    return data.join("&");
  }

      var body = formToEncodedBody(form);
      var url = form.getAttribute("action") || window.location.href.split("#")[0];

      var xhr = new XMLHttpRequest();
      xhr.open("POST", url, true);
      xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded; charset=UTF-8");
      xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");

      xhr.onreadystatechange = function () {
        if (xhr.readyState !== 4) {
          return;
        }

        setButtonState(submitBtn, false);

        var parsed = null;

        try {
          parsed = JSON.parse(xhr.responseText || "{}");
        } catch (e) {
          parsed = null;

        if (xhr.status >= 200 && xhr.status < 400 && parsed && parsed.success === true) {
          showSaveToast(parsed.message || "Saved successfully.", "success");
          showInlineSaveStatus(form, parsed.message || "Saved successfully.");
          return;
        }

        var message = "Save failed. Please refresh the page and try again.";

        if (parsed && parsed.message) {
          message = parsed.message;
        }

        showSaveToast(message, "error");
      };

      xhr.onerror = function () {
        setButtonState(submitBtn, false);
        showSaveToast("Network error while saving. Please try again.", "error");
      };

      try {
        xhr.send(body);
      } catch (e) {
        setButtonState(submitBtn, false);

      }

      return false;
    }, true);
  }

    }
    var combo = BATCH_COMBOS[batchIndex];
    setCell(combo.profile, combo.strategy, "Running...", "running");
    setCurrentLabel(combo.label);
    var idx = batchIndex;
    batchIndex += 1;

    runCombo(combo, function (res) {
      if (batchStopped) {
        onBatchEnd();
        return;
      }
      batchResults[idx] = { combo: combo, html: res.html, text: res.text, done: true,
                            mainNumbers: res.mainNumbers || [], extraNumbers: res.extraNumbers || [] };
      persistResults();
      batchDoneCount += 1;
      updateProgress();
      var viewHtml = "<span style='color:#16a34a;font-weight:600;'>Done</span> <a href='#' style='font-size:0.8rem;margin-left:4px;' data-batch-view='" + idx + "'>View</a>";
      setCell(combo.profile, combo.strategy, viewHtml, "done");
      setTimeout(runNext, 1800); /* 1.8-second cooldown between combo runs */
    });
  }

  function onBatchEnd() {
    batchRunning = false;
    setCurrentLabel(batchStopped ? "Batch stopped." : "All runs complete.");
    try { console.log("[SKAI batch] onBatchEnd: " + batchDoneCount + "/" + BATCH_COMBOS.length + " combos done, stopped=" + batchStopped); } catch (_) {}
    var saveRow = qs("skai-batch-save-row");
    if (saveRow && batchDoneCount > 0 && (batchStopped || batchDoneCount >= BATCH_COMBOS.length)) {
      saveRow.style.display = "flex";
    }
    var stopBtn = qs("skai-batch-stop");
    if (stopBtn) { stopBtn.disabled = true; }
  }

  function resetGrid() {
    var cells = document.querySelectorAll("#skai-batch-grid td[data-profile]");
    for (var i = 0; i < cells.length; i++) {
      cells[i].innerHTML = "Pending";
      cells[i].style.background = "";
    }
  }

  function openDetail(idx) {
    var r = batchResults[idx];
    if (!r) { return; }
    var detail = qs("skai-batch-detail");
    var title = qs("skai-batch-detail-title");
    var body = qs("skai-batch-detail-body");
    if (!detail || !body) { return; }
    if (title) { title.textContent = r.combo.label; }
    body.textContent = r.text || "(no result text)";
    detail.style.display = "block";
    try { detail.scrollIntoView({ behavior: "smooth", block: "nearest" }); } catch (_) {}
  }

  function saveAllSequential(indices, doneCallback, saveCount, totalCount) {
    if (saveCount === undefined) { saveCount = 0; }
    if (totalCount === undefined) { totalCount = indices.length; }
    if (!indices || indices.length === 0) {
      if (doneCallback) { doneCallback(); }
      return;
    }
    var idx = indices[0];
    var rest = indices.slice(1);
    var r = batchResults[idx];
    if (!r || !r.done) {
      saveAllSequential(rest, doneCallback, saveCount, totalCount);
      return;
    }

    try { selectProfile(r.combo.profile); } catch (_) {}
    try { selectStrategy(r.combo.strategy); } catch (_) {}

    var form = qs("savePredictionForm");
    if (!form) {
      saveAllSequential(rest, doneCallback, saveCount, totalCount);
      return;
    }

    try {
      if (r.mainNumbers && r.mainNumbers.length) {
        window.skaiSaveMainOrdered  = r.mainNumbers.slice();
        window.skaiSaveExtraOrdered = (r.extraNumbers && r.extraNumbers.length)
          ? r.extraNumbers.slice() : [];
        var mainField = document.getElementById("saveMainNumbers");
        if (mainField) { mainField.value = r.mainNumbers.join(","); }
        var extraField = document.getElementById("saveExtraBallNumbers");
        if (extraField) { extraField.value = (r.extraNumbers || []).join(","); }
      }
      /* Update label to include combo name so each save is distinct */
      var LABEL_DELIMITER = " | ";
      var labelField = document.getElementById("saveLabel");
      if (labelField && r.combo && r.combo.label) {
        var baseLbl = labelField.value || "Prediction";
        /* Strip any previous combo suffix (format: "... | Combo Name") */
        var pipeIdx = baseLbl.indexOf(LABEL_DELIMITER);
        if (pipeIdx >= 0) { baseLbl = baseLbl.substring(0, pipeIdx); }
        labelField.value = baseLbl + LABEL_DELIMITER + r.combo.label;
      }
    } catch (_) {}

    /* Show saving progress: Saving 1 of 9, 2 of 9 ... */
    var currentSaveNum = saveCount + 1;
    if (window.skaiShowSaveToast) {
      window.skaiShowSaveToast("Saving " + currentSaveNum + " of " + totalCount + "...");
    }

    var strategyToMode = { ai: "ai-forward", hybrid: "mixed", skip: "skip-dominant" };
    var modeField = document.getElementById("saveModeKey");
    if (modeField && r.combo && r.combo.strategy) {
      modeField.value = strategyToMode[r.combo.strategy] || "balanced";
    }
    var runField = document.getElementById("saveRunUuid");
    if (runField) { runField.value = ""; }
    var hashField = document.getElementById("saveClientResultHash");
    if (hashField) { hashField.value = ""; }

    function submitPreparedBatchSave() {
      var ev = document.createEvent("Event");
      ev.initEvent("submit", true, true);
      form.dispatchEvent(ev);

      setTimeout(function () {
        saveAllSequential(rest, doneCallback, currentSaveNum, totalCount);
      }, 2500); /* 2.5-second delay between sequential saves */
    }

    if (typeof window.skaiPrepareClientIntegrity === "function") {
      window.skaiPrepareClientIntegrity().then(submitPreparedBatchSave).catch(function () {
        if (window.skaiShowSaveToast) {
          window.skaiShowSaveToast("Could not prepare secure save for this batch run.", "error");
        }
        setTimeout(function () {
          saveAllSequential(rest, doneCallback, currentSaveNum, totalCount);
        }, 1200);
      });
    } else {
      submitPreparedBatchSave();
    }
  }

  function wireButtons() {
    var launchBtn = qs("skai-batch-launch");
    var modal = qs("skai-batch-modal");
    var confirmBtn = qs("skai-batch-confirm");
    var cancelBtn = qs("skai-batch-cancel");
    var stopBtn = qs("skai-batch-stop");
    var clearBtn = qs("skai-batch-clear");
    var detailClose = qs("skai-batch-detail-close");
    var saveAllBtn = qs("skai-batch-save-all");
    var saveSelBtn = qs("skai-batch-save-selected");

    if (launchBtn) {
      launchBtn.addEventListener("click", function () {
        if (modal) { modal.style.display = "flex"; }
      }, false);
    }

    if (cancelBtn) {
      cancelBtn.addEventListener("click", function () {
        if (modal) { modal.style.display = "none"; }
      }, false);
    }

    if (modal) {
      modal.addEventListener("click", function (e) {
        if (e.target === modal) { modal.style.display = "none"; }
      }, false);
    }

    if (confirmBtn) {
      confirmBtn.addEventListener("click", function () {
        hideModal();
        if (batchRunning) { return; }
        batchRunning = true;
        batchStopped = false;
        batchIndex = 0;
        batchDoneCount = 0;
        batchResults = [];
        var saveRow = qs("skai-batch-save-row");
        if (saveRow) { saveRow.style.display = "none"; }
        var stopBtn2 = qs("skai-batch-stop");
        if (stopBtn2) { stopBtn2.disabled = false; }
        var detailEl = qs("skai-batch-detail");
        if (detailEl) { detailEl.style.display = "none"; }
        resetGrid();
        showPanel();
        updateProgress();
        setCurrentLabel("Starting...");
        setTimeout(runNext, 400);
      }, false);
    }

    if (stopBtn) {
      stopBtn.addEventListener("click", function () {
        batchStopped = true;
        batchRunning = false;
        setCurrentLabel("Stopping...");
      }, false);
    }

    if (clearBtn) {
      clearBtn.addEventListener("click", function () {
        batchStopped = true;
        batchRunning = false;
        batchResults = [];
        batchDoneCount = 0;
        batchIndex = 0;
        try { sessionStorage.removeItem(STORAGE_KEY); } catch (_) {}
        resetGrid();
        updateProgress();
        setCurrentLabel("Ready");
        var detail2 = qs("skai-batch-detail");
        if (detail2) { detail2.style.display = "none"; }
        var saveRow2 = qs("skai-batch-save-row");
        if (saveRow2) { saveRow2.style.display = "none"; }
      }, false);
    }

    if (detailClose) {
      detailClose.addEventListener("click", function () {
        var d = qs("skai-batch-detail");
        if (d) { d.style.display = "none"; }
      }, false);
    }

    var grid = qs("skai-batch-grid");
    if (grid) {
      grid.addEventListener("click", function (e) {
        var t = e.target;
        if (t && t.getAttribute("data-batch-view") !== null) {
          try { e.preventDefault(); } catch (_) {}
          openDetail(parseInt(t.getAttribute("data-batch-view"), 10));
        }
      }, false);
    }

    if (saveAllBtn) {
      saveAllBtn.addEventListener("click", function () {
        if (!window.confirm("This will save each completed run one by one to your dashboard. Continue?")) { return; }
        var indices = [];
        for (var i = 0; i < batchResults.length; i++) {
          if (batchResults[i] && batchResults[i].done) { indices.push(i); }
        }
        if (indices.length === 0) {
          alert("No completed runs to save.");
          return;
        }
        var totalRuns = indices.length;
        saveAllSequential(indices, function () {
          if (window.skaiShowSaveToast) { window.skaiShowSaveToast("All " + totalRuns + " " + (totalRuns === 1 ? "run" : "runs") + " saved to your dashboard."); }
        });
      }, false);
    }

    if (saveSelBtn) {
      saveSelBtn.addEventListener("click", function () {
        var available = [];
        for (var i = 0; i < batchResults.length; i++) {
          if (batchResults[i] && batchResults[i].done) { available.push(i); }
        }
        if (available.length === 0) {
          alert("No completed runs to save.");
          return;
        }
        var labels = [];
        for (var j = 0; j < available.length; j++) {
          labels.push((j + 1) + ". " + batchResults[available[j]].combo.label);
        }
        var input = window.prompt(
          "Enter the numbers (comma-separated) of the runs you want to save:\n\n" + labels.join("\n"),
          "1"
        );
        if (!input) { return; }
        var chosen = [];
        var parts = input.split(",");
        for (var k = 0; k < parts.length; k++) {
          var n = parseInt(parts[k].trim(), 10);
          if (!isNaN(n) && n >= 1 && n <= available.length) {
            chosen.push(available[n - 1]);
          }
        }
        if (chosen.length === 0) { alert("No valid selections."); return; }
        saveAllSequential(chosen, function () {
          if (window.skaiShowSaveToast) { window.skaiShowSaveToast("Selected runs saved."); }
        });
      }, false);
    }
  }

  function init() {
    wireButtons();
    try { sessionStorage.removeItem(STORAGE_KEY); } catch (_) {}
  }

  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", init, {once: true});
  } else {
    init();
  }
})();
